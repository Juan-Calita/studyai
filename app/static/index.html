function pollStatus(id) { const statusMsgs = { 'processando': {p: 15, m: 'Preparando áudio...'}, 'transcrevendo': {p: 30, m: 'Transcrevendo áudio...'}, 'estruturando': {p: 55, m: 'Transcrição pronta! Gerando resumo expandido, guia de estudos e flashcards...'}, 'gerando_conteudo': {p: 70, m: 'Gerando resumo expandido, guia de estudos e flashcards...'}, 'gerando_arquivos': {p: 90, m: 'Gerando PDF e deck Anki...'}, }; let currentProgress = 10; let targetProgress = 15; let currentMsg = 'Preparando áudio...'; // Animação suave da barra const smoothAnimation = setInterval(() => { if (currentProgress < targetProgress) { currentProgress += 1; setProgress(currentProgress, currentMsg); } }, 400); const interval = setInterval(async () => { try { const res = await fetch(API + '/api/aulas/' + id); const data = await res.json(); // Atualiza progresso baseado no status real if (statusMsgs[data.status]) { targetProgress = statusMsgs[data.status].p; currentMsg = statusMsgs[data.status].m; } if (data.status === 'pronto') { clearInterval(interval); clearInterval(smoothAnimation); setProgress(100, 'Concluído!'); aulaData = data; setTimeout(() => renderApp(), 600); } else if (data.status === 'erro') { clearInterval(interval); clearInterval(smoothAnimation); showError('Erro: ' + (data.erro || 'desconhecido')); } } catch(_){} }, 2000); }
